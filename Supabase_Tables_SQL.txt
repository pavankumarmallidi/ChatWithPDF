create table public."PDF_DATA_INFO" (
  id bigint generated by default as identity not null,
  created_at timestamp with time zone not null default now(),
  "EMAIL ID" character varying not null,
  "PDF NAME" character varying not null,
  "PDF SUMMARY" character varying not null,
  "PAGES" bigint not null,
  "WORDS" bigint not null,
  "LANGUAGE" text not null,
  "OCR OF PDF" text not null,
  constraint PDF_DATA_INFO_pkey primary key (id)
) TABLESPACE pg_default;
create table public."CHAT_HISTORY" (
  id bigint generated by default as identity not null,
  created_at timestamp with time zone not null default now(),
  "CHAT_ID" bigint not null,
  "EMAIL_ID" text not null,
  "SENDER" text not null,
  "RECIVER" text not null,
  "MESSAGE" text null,
  pdf_ids json[] not null,
  constraint CHAT_HISTORY_pkey primary key (id)
) TABLESPACE pg_default;
